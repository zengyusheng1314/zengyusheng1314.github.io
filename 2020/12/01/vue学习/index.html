    <div id="app">

            <li>

            </li>
            `,
            data: function() {
                return {}
            },

            },
        })

            template: `

            props: {
                title: String,
                del: {
                    type: Boolean,

            template: ` 
             <li>
                <span v-if="!del">{{title}}</span>

插槽

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
</head>
<body>
    <div id="app">
        <todo-list>
            <!-- <template v-slot>    默认插槽   -->
            <todo-item @delete="handleDelete" v-for="item in list" :title="item.title" :del="item.del">
                <!-- vue.js 2.6语法  -->
                <template v-slot:pre-slot="{value}">
                    <!-- 本质上是返回组件的函数 可根据不同的值传递不同的内容 -->
                       <span>前置{{value}}</span>
                    </template>
                <template v-slot:suf-slot>
                        <span>前置</span>
                     </template>
                <!--  vue.js 2.5语法
                        <span slot="pre-slot"> 前置</span>
                    <span slot="suf-slot">后置</span> -->
            </todo-item>
            <!-- </template> -->
        </todo-list>
    </div>
    <!-- CDN 引入vue-->
    <script src="https://cdn.jsdelivr.net/npm/vue/dist/vue.js"></script>
    <script>
        // 创建一个组件
        Vue.component('todo-item', {
            // 属性声明
            props: {
                title: String,
                del: {
                    type: Boolean,
                    default: false, //默认值
                },
            },
            // 模板   插槽标签 slot
            template: ` 
             <li>
                <slot name="pre-slot"  :value="value"></slot>
                <span v-if="!del">{{title}}</span>
                <span v-else style="text-decoration: line-through">{{title}}</span>
                <slot name="suf-slot"></slot>
                <button v-show="!del" 
                @click="handleClick">删除</button>
            </li>
            `,
            data: function() {
                return {
                    value: Math.random()
                }
            },
            methods: {
                handleClick() {
                    console.log('点击删除按钮')
                    this.$emit('delete', this.title)
                }
            },
        })
        Vue.component('todo-list', {
            // 插槽标签 slot
            template: `
            <ul>
               <slot></slot>
        </ul>
            `,
            data: function() {
                return {
                    list: [{
                        title: '课程4',
                        del: false,
                    }, {
                        title: '课程5',
                        del: true,
                    }],
                }
            },

        })
        var vm = new Vue({
            el: '#app',
            data: {
                message: 'hello world',
                list: [{
                    title: '课程2',
                    del: false,
                }, {
                    title: '课程3',
                    del: true,
                }],
            },
            methods: {
                handleDelete(val) {
                    console.log('handleDelete', val)
                }
            }
        })
    </script>
</body>
</html>

双向绑定和单向数据流

<!-- 数据的双向绑定本质上是单向数据流
v-model 在内部为不同的输入元素使用不同的 property 并抛出不同的事件：
text 和 textarea 元素使用 value property 和 input 事件；
checkbox 和 radio 使用 checked property 和 change 事件；
select 字段将 value 作为 prop 并将 change 作为事件。
<template>
-->
  <div id="app">
        <TodoList>
           <input v-model="message">
           {{message}}
           <input :value="message" @input="handleChange">  
  </div>
</template>
<script>
export default {
  name: 'App',
  data() {
    return { }
  },
  methods: {
        handleChange(e){
            this.message=e.target.value
        }
  },
}
</script>
